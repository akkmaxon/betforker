def parsed_bookies data
  open(@log_file, 'a') do |f|
    f.write("\n############################\n")
    data.each do |buk|
      buk.each do |key, val|
        f.write(" #{key}:  #{val}\n")
      end
    end
    f.close
  end
end

def to_screen forks_to_screen
  to_screen = ""
  forks_to_screen.each do |fork|
    fork.each do |key, val|
      if key == :bookies
        to_screen << "#{val}\n" unless to_screen.include?(val)
      else
        to_screen << "#{val}  "
      end
    end
    to_screen += "\n"
  end
  case ENV['XDG_CURRENT_DESKTOP']
  when 'KDE'
    %x`kdialog --passivepopup "#{to_screen}" 30`
  when 'XFCE'
    %x`notify-send -t 30000 "#{to_screen}"`
  when 'GNOME'
    puts "GNOME output not ready yet"
  else
    puts to_screen
  end
end
